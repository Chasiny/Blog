class Solution {
public:
    ListNode* Merge(ListNode* pHead1, ListNode* pHead2)
    {
        if(pHead1==NULL){
            return pHead2;
        }
        if(pHead2==NULL){
            return pHead1;
        }
        ListNode* small=pHead1->val<pHead2->val?pHead1:pHead2;
        ListNode* big=pHead1->val>=pHead2->val?pHead1:pHead2;
        ListNode* pos=small;
        while(pos->next){
            while(big&&big->val<pos->next->val){
                ListNode* temp=big->next;
                big->next=pos->next;
                pos->next=big;
                big=temp;
            }
            pos=pos->next;
        }
        if(big!=NULL){
            pos->next=big;
        }
        return small;
    }
};
